<script>

let currentDate=todayKey()


function todayKey(){
return new Date().toISOString().slice(0,10)
}



function saveDay(){

let data={

morning:morning.checked,
lunch:lunch.checked,
snack:snack.checked,
dinner:dinner.checked,
evening:evening.checked,

mfood:mfood.value,
lfood:lfood.value,
sfood:sfood.value,
dfood:dfood.value,
efood:efood.value

}

localStorage.setItem(currentDate,JSON.stringify(data))

loadEverything()

}



function loadDay(date){

currentDate=date

let d=JSON.parse(localStorage.getItem(date))

if(!d) d={}

morning.checked=d.morning||false
lunch.checked=d.lunch||false
snack.checked=d.snack||false
dinner.checked=d.dinner||false
evening.checked=d.evening||false

mfood.value=d.mfood||""
lfood.value=d.lfood||""
sfood.value=d.sfood||""
dfood.value=d.dfood||""
efood.value=d.efood||""

dateTitle.innerHTML=date

}



function loadHistory(){

history.innerHTML=""

Object.keys(localStorage)
.sort()
.reverse()
.forEach(key=>{

let div=document.createElement("div")

div.className="entry"

div.innerHTML=key

div.onclick=()=>loadDay(key)

history.appendChild(div)

})

}



function loadFoodLog(){

foodLog.innerHTML=""

Object.keys(localStorage)
.sort()
.reverse()
.forEach(key=>{

let d=JSON.parse(localStorage[key]) || {}

let div=document.createElement("div")

div.className="entry"

div.innerHTML=

"<b>"+key+"</b><br><br>" +

"Morning: " + (d.mfood || "") + "<br>" +

"Lunch: " + (d.lfood || "") + "<br>" +

"Snack: " + (d.sfood || "") + "<br>" +

"Dinner: " + (d.dfood || "") + "<br>" +

"Evening: " + (d.efood || "")

foodLog.appendChild(div)

})
}



function getScore(d){

return(
(d.morning?1:0)+
(d.lunch?1:0)+
(d.snack?1:0)+
(d.dinner?1:0)+
(d.evening?1:0)
)

}



function loadStreak(){

let days=Object.keys(localStorage).sort().reverse()

let streak=0

for(let k of days){

let d=JSON.parse(localStorage[k])

if(getScore(d)>=3) streak++
else break

}

streakDiv.innerHTML="🔥 "+streak+" Day Streak"

}



function loadChart(){

let keys=Object.keys(localStorage).sort()

let scores=[]

keys.forEach(k=>{
scores.push(getScore(JSON.parse(localStorage[k])))
})

new Chart(chart,{

type:"line",

data:{
labels:keys,
datasets:[{label:"Score",data:scores}]
}

})

}



function loadCalendar(){

calendar.innerHTML=""

Object.keys(localStorage)
.sort()
.reverse()
.slice(0,30)
.forEach(k=>{

let div=document.createElement("div")

div.className="entry"

div.innerHTML=k

div.onclick=()=>loadDay(k)

calendar.appendChild(div)

})

}



function toggleDark(){

document.body.classList.toggle("dark")

}



function exportData(){

let data=JSON.stringify(localStorage)

let blob=new Blob([data])

let a=document.createElement("a")

a.href=URL.createObjectURL(blob)

a.download="gentle-eating-backup.json"

a.click()

}



function importData(){

let file=document.getElementById("importFile").files[0]

let reader=new FileReader()

reader.onload=function(){

let data=JSON.parse(reader.result)

for(let k in data)
localStorage.setItem(k,data[k])

loadEverything()

}

reader.readAsText(file)

}



function loadEverything(){

loadHistory()
loadFoodLog()
loadCalendar()
loadChart()
loadStreak()

loadDay(currentDate)

}


loadEverything()
